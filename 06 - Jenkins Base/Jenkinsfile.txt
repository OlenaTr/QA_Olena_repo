pipeline {
  agent any
  
    environment {
        POSTMAN_API_KEY = credentials("postman-api-key")
    }

  stages {

    stage('Postman CLI Login') {
      steps {
        sh 'postman login --with-api-key $POSTMAN_API_KEY'
        }
    }

    stage('Running collection') {
      steps {
        sh 'postman collection run "YOUR COLLECTION ID"'
      }
    }
  }
}
